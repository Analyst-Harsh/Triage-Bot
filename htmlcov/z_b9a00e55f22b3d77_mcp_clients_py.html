17def build_mcp_connections(settings: Settings) -> dict[str, Connection]: 

18 """Servers configured this run — `github` when a token is set, `docmind` 

19 only when its (still-unfinalized) launch command is configured. Neither 

20 is required: an unconfigured server is logged and simply absent from the 

21 returned mapping, not an error — the Researcher degrades gracefully.""" 

22 connections: dict[str, Connection] = {} 

23 

24 if settings.github_token is not None: 

25 connections["github"] = { 

26 "transport": "streamable_http", 

27 "url": settings.github_mcp_url, 

28 "headers": {"Authorization": f"Bearer {settings.github_token.get_secret_value()}"}, 

29 } 

30 else: 

31 log.warning("researcher_tool_unavailable", tool="github", reason="GITHUB_TOKEN not set") 

32 

33 if settings.docmind_mcp_command is not None: 

34 connections["docmind"] = { 

35 "transport": "stdio", 

36 "command": settings.docmind_mcp_command, 

37 "args": settings.docmind_mcp_args, 

38 "cwd": settings.docmind_mcp_cwd, 

39 } 

40 else: 

41 log.warning( 

42 "researcher_tool_unavailable", 

43 tool="docmind", 

44 reason="DOCMIND_MCP_COMMAND not set", 

45 ) 

46 

47 return connections

50def clamp_tool_output(tool: BaseTool, max_chars: int = 8_000) -> BaseTool: 

51 """Wraps `tool` so its output is clamped to `max_chars`, with the 

52 truncation noted inline so the model knows the excerpt is partial. 

53 

54 Tool outputs are untrusted input — issue text, GitHub comments, and web 

55 pages are all attacker- or at least author-controlled — so this bounds 

56 context/cost blowup from a single huge file or webpage. Delegates to the 

57 original tool's own `ainvoke()` rather than touching its internals, so 

58 it works uniformly across MCP-provided and directly-constructed tools. 

59 

60 `max_chars` defaults to a literal here (not `Settings`) so this stays a 

61 pure, standalone-testable helper; `researcher_toolset` is the one 

62 production call site and always passes `settings.researcher_tool_output_max_chars` 

63 explicitly. 

64 """ 

65 

66 async def _clamped(**kwargs: object) -> str: 

67 result = await tool.ainvoke(kwargs) 

68 text = result if isinstance(result, str) else str(result) 

69 if len(text) <= max_chars: 

70 return text 

71 omitted = len(text) - max_chars 

72 return f"{text[:max_chars]}\n...[truncated, {omitted} more characters]" 

73 

74 coroutine: Callable[..., Awaitable[str]] = _clamped 

75 return StructuredTool.from_function( 

76 coroutine=coroutine, 

77 name=tool.name, 

78 description=tool.description, 

79 args_schema=tool.args_schema, 

80 )

83@asynccontextmanager 

84async def researcher_toolset(settings: Settings) -> AsyncGenerator[list[BaseTool]]: 

85 """Loads the Researcher's tools for the graph's lifetime. 

86 

87 Opens **persistent** MCP sessions (`client.session()` held open via an 

88 `AsyncExitStack`, not the ephemeral per-call default) — for the stdio 

89 DocMind server, the default mode would respawn its process on every 

90 single tool call. Sessions close on exit, mirroring 

91 `graph/checkpointer.py`'s `sqlite_checkpointer()` context-manager idiom. 

92 

93 Each unavailable/failed tool is logged and omitted, never fatal — an 

94 empty toolset still runs the Researcher (low-confidence, issue-text-only 

95 findings), which keeps replay/dev usable without any secrets configured. 

96 """ 

97 connections = build_mcp_connections(settings) 

98 tools: list[BaseTool] = [] 

99 

100 async with AsyncExitStack() as stack: 

101 if connections: 

102 client = MultiServerMCPClient(connections) 

103 for server_name in connections: 

104 try: 

105 session = await stack.enter_async_context(client.session(server_name)) 

106 server_tools = await load_mcp_tools(session) 

107 except Exception: 

108 log.warning("researcher_tool_load_failed", server=server_name, exc_info=True) 

109 continue 

110 tools.extend( 

111 clamp_tool_output(tool, max_chars=settings.researcher_tool_output_max_chars) 

112 for tool in server_tools 

113 ) 

114 

115 if settings.tavily_api_key is not None: 

116 web_tool: Any = TavilySearch(tavily_api_key=settings.tavily_api_key) 

117 tools.append( 

118 clamp_tool_output(web_tool, max_chars=settings.researcher_tool_output_max_chars) 

119 ) 

120 else: 

121 log.warning("researcher_tool_unavailable", tool="web", reason="TAVILY_API_KEY not set") 

122 

123 yield tools
