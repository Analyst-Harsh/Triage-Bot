36class AgentLoopState(TriageState): 

37 """`TriageState` plus private trajectory-bridging channels. None of 

38 these keys exist on `TriageState` itself, so they start empty/absent 

39 every run and never propagate back to the parent graph — each agent-loop 

40 node owns its own trajectory, with no cross-node scoping to manage (see 

41 the "Messages scoping" note in `AGENTS.md`). 

42 

43 `summary` is typed loosely (`BaseModel | None`) rather than generically: 

44 it's purely internal bridging between this module's own `_summarize_node` 

45 and `_assemble_node`, never seen by subclass authors directly — they get 

46 the properly-typed `SummaryT` via `finalize()`'s signature instead (see 

47 the `cast` in `_assemble_node`). 

48 """ 

49 

50 messages: Annotated[list[BaseMessage], add_messages] 

51 summary: BaseModel | None 

52 summarize_cost: float

64class AgentSubgraph[SummaryT: BaseModel](ABC): 

65 """Subgraph-level counterpart to `TriageNode`, for nodes whose logic is 

66 an LLM tool-calling loop rather than a single call. `TriageNode` can't be 

67 reused directly here: LangGraph's subgraph detection (checkpoint 

68 namespacing, nested streaming) requires the loop to be its own compiled 

69 `StateGraph`, registered directly via `add_node()` — wrapping a 

70 `.invoke()` call inside a `TriageNode.execute()` body would silently 

71 defeat that detection (see `TriageNode`'s docstring in `base.py`). So 

72 this class's product is a compiled graph, not an executed result: 

73 `compile()` returns a `CompiledStateGraph` for the caller to 

74 `add_node(name, ...)` directly, exactly like a plain `TriageNode` 

75 instance. 

76 

77 Concrete subclasses supply only two hooks — `prepare` (build the initial 

78 trajectory message, or short-circuit) and `finalize` (map the parsed 

79 summary into the node's own output slot). Everything else — the 

80 tool-call cap, the post-loop structured summary call, run-meta 

81 bookkeeping, structured logging — is uniform machinery on this base, 

82 mirroring how `TriageNode.__call__` is uniform over every simple node. 

83 """ 

84 

85 name: ClassVar[NodeName] 

86 llm_config: ClassVar[NodeLLMConfig] 

87 max_tool_calls: ClassVar[int] 

88 summary_schema: ClassVar[type[BaseModel]] 

89 

90 # Every in-loop model call resends the full accumulated trajectory, 

91 # ToolMessage content included -- with no bound on cumulative size, 

92 # tool-heavy runs (the Drafter's sandbox loop) can grow into TPM 

93 # rate-limit territory after 8-10+ tool calls. These three tune 

94 # ContextEditingMiddleware/ClearToolUsesEdit (see build_agent) rather 

95 # than living in Settings: they're algorithmic constants tied to this 

96 # node's own tool-output budgets, the same category as 

97 # DRAFTER_MAX_TOOL_CALLS/RESEARCHER_MAX_TOOL_CALLS, not "ops-tunable 

98 # infra values" (Settings' own docstring scope). 

99 # 

100 # trigger=20_000 (vs. the library default 100_000): drafter_file_read_max_chars 

101 # =16_000 chars is ~4,000-4,800 tokens per read_file call, so 20,000 

102 # engages proactively around the observed 8-10-call inflection point 

103 # rather than reactively near a 30-call worst case. 

104 context_edit_trigger_tokens: ClassVar[int] = 20_000 

105 # keep=6 (vs. library default 3): DRAFTER_MAX_TOOL_CALLS's own budget 

106 # comment counts ~4 tool calls per fix cycle (read+edit+run) -- keeping 

107 # 6 preserves roughly the current + previous cycle's outputs verbatim, 

108 # so a model that just edit_file'd and wants to read_file its own diff 

109 # doesn't find that exact result already cleared (costs one extra tool 

110 # call if it does, well inside budget -- not a correctness bug). 

111 context_edit_keep_tool_results: ClassVar[int] = 6 

112 context_edit_placeholder: ClassVar[str] = ( 

113 "[tool output cleared to manage context size -- call this tool " 

114 "again if you need the original data]" 

115 )

117 def __init__(self, tools: list[BaseTool]) -> None: 

118 """Tools are injected (loaded by the composition root — see 

119 `tools/mcp_clients.py`), not constructed here: graph/subgraph 

120 construction must stay side-effect-free, and MCP tool loading is 

121 inherently I/O.""" 

122 settings = get_settings() 

123 self._tools = tools 

124 self._settings = settings 

125 self._primary_model = create_chat_model(self.llm_config.primary, settings) 

126 self._fallback_model = create_chat_model(self.llm_config.fallback, settings)

128 def system_prompt(self) -> str | None: 

129 """Static system prompt for the tool-calling loop, built from 

130 `self._tools` (already loaded by `__init__`) so it can name what's 

131 actually available this run. `None` (the default) means the model's 

132 own default system behavior — override when a node needs to frame 

133 its task and tool set explicitly, which every real subclass does.""" 

134 return None

136 @abstractmethod 

137 def prepare(self, state: TriageState) -> list[BaseMessage] | None: 

138 """Build the initial trajectory message(s) for this run, or return 

139 `None` to short-circuit: skip the tool-calling loop *and* the 

140 `summarize` LLM call entirely (e.g. the Planner already decided no 

141 investigation is warranted) — `finalize` then receives `summary=None` 

142 and must build minimal output programmatically, at zero LLM/tool 

143 spend.""" 

144 raise NotImplementedError 

145 

146 @abstractmethod 

147 async def finalize( 

148 self, 

149 summary: SummaryT | None, 

150 tool_calls: list[ToolCallRecord], 

151 state: TriageState, 

152 ) -> TriageStateUpdate: 

153 """Map the parsed summary (or `None`, on the short-circuit path) plus 

154 the derived tool-call records into this node's output slot and 

155 `status`. Run-meta bookkeeping (cost, `tool_calls_made`, 

156 `iteration_count`) is applied by `_assemble_node` after this 

157 returns — don't set `run_meta` here unless accumulating your own 

158 cost on top of it first; a `run_meta` update this method does set is 

159 still respected as the base `_assemble_node` accumulates onto (same 

160 contract as `TriageNode.execute()`). 

161 

162 Async so a subclass can make its own independent LLM call here — e.g. 

163 the Drafter's grounding self-check, a second structured-output call 

164 genuinely separate from the one that produced `summary` (an 

165 evaluator-optimizer pattern: a model is a much weaker judge of its 

166 own claims in the same breath it wrote them). Subclasses that need no 

167 I/O here (the common case) just don't `await` anything — a 

168 no-await async function behaves identically to a sync one.""" 

169 raise NotImplementedError

171 def compile(self) -> CompiledStateGraph[AgentLoopState, None, AgentLoopState, AgentLoopState]: 

172 """Returns a compiled graph over `AgentLoopState`, for the caller to 

173 `add_node(name, ...)` directly into a `TriageState`-schema parent 

174 graph. Only keys the two schemas share (`issue`, `planner_output`, 

175 `research_findings`, ..., `run_meta`) flow across that boundary in 

176 either direction — LangGraph's subgraph nesting matches state by key 

177 name, not by declared schema type, so `messages`/`summary`/ 

178 `summarize_cost` here stay private to this subgraph.""" 

179 graph = StateGraph(AgentLoopState) 

180 graph.add_node("prepare", self.prepare_node) # pyright: ignore[reportUnknownMemberType] 

181 graph.add_node("agent", self.build_agent()) # pyright: ignore[reportUnknownMemberType] 

182 graph.add_node("summarize", self.summarize_node) # pyright: ignore[reportUnknownMemberType] 

183 graph.add_node("assemble", self.assemble_node) # pyright: ignore[reportUnknownMemberType] 

184 graph.add_edge(START, "prepare") 

185 graph.add_conditional_edges( # pyright: ignore[reportUnknownMemberType] 

186 "prepare", self.route_after_prepare, {"agent": "agent", "assemble": "assemble"} 

187 ) 

188 graph.add_edge("agent", "summarize") 

189 graph.add_edge("summarize", "assemble") 

190 graph.add_edge("assemble", END) 

191 return graph.compile() # pyright: ignore[reportUnknownMemberType]

193 def prepare_node(self, state: AgentLoopState) -> _LoopUpdate: 

194 """Graph-node wrapper around `prepare()`. Not underscore-prefixed: 

195 it's the template-method contact point tests exercise directly (see 

196 `tests/graph/nodes/test_agent_subgraph.py`), the same way 

197 `TriageNode.execute()` is a public override point despite existing 

198 purely to be called by graph wiring.""" 

199 log.info("agent_subgraph_started", node=self.name) 

200 initial_messages = self.prepare(state) 

201 if initial_messages is None: 

202 return _LoopUpdate() 

203 return _LoopUpdate(messages=initial_messages)

206 return "agent" if state["messages"] else "assemble"

208 def _middleware( 

209 self, 

210 ) -> list[ContextEditingMiddleware | ToolCallLimitMiddleware | ModelFallbackMiddleware]: 

211 """`ContextEditingMiddleware` first (outermost): `ToolCallLimitMiddleware` 

212 only implements `after_model`, never `wrap_model_call`, so its position 

213 relative to the other two is inert either way. `ModelFallbackMiddleware` 

214 does participate in `wrap_model_call` -- placing the context edit 

215 outermost means its (deterministic, cheap) clearing pass is computed 

216 once per turn and reused by any same-turn fallback retry, rather than 

217 recomputed per attempt. 

218 

219 `ContextEditingMiddleware.wrap_model_call` only edits the ephemeral 

220 outgoing request for one model call -- it never returns a state 

221 update, so it never touches the persisted `AgentLoopState.messages` 

222 channel (verified by reading the installed langchain source directly). 

223 `summarize_node`'s own structured-output call isn't routed through 

224 this middleware at all, which is why it applies the same 

225 `ClearToolUsesEdit` logic itself, via `clamp_trajectory_for_model_call`.""" 

226 return [ 

227 ContextEditingMiddleware( 

228 edits=( 

229 ClearToolUsesEdit( 

230 trigger=self.context_edit_trigger_tokens, 

231 keep=self.context_edit_keep_tool_results, 

232 placeholder=self.context_edit_placeholder, 

233 ), 

234 ), 

235 ), 

236 ToolCallLimitMiddleware(run_limit=self.max_tool_calls, exit_behavior="end"), 

237 ModelFallbackMiddleware(self._fallback_model), 

238 ]

255 async def summarize_node(self, state: AgentLoopState) -> _LoopUpdate: 

256 # A parallel tool-call batch that straddles the cap can leave some 

257 # "allowed" calls counted by ToolCallLimitMiddleware but never 

258 # actually run, with no ToolMessage — patch those in (transiently, 

259 # correctly ordered) before this trajectory goes back to a model, 

260 # or the call below fails outright on the unresolved tool_call. See 

261 # `resolve_dangling_tool_calls`'s docstring for why this can't just 

262 # be appended via the `messages` channel's reducer. 

263 messages = resolve_dangling_tool_calls(state["messages"]) 

264 # This call sits outside build_agent()'s create_agent loop, so 

265 # ContextEditingMiddleware never runs over it -- apply the same 

266 # deterministic clearing pass directly, on a throwaway copy, so 

267 # this one structured-output call doesn't resend the full, 

268 # unpruned trajectory. state["messages"] itself is untouched: the 

269 # checkpoint stays a faithful record (see 

270 # resolve_dangling_tool_calls's own docstring for the same 

271 # precedent). 

272 clamped_messages = clamp_trajectory_for_model_call( 

273 messages, 

274 trigger=self.context_edit_trigger_tokens, 

275 keep=self.context_edit_keep_tool_results, 

276 placeholder=self.context_edit_placeholder, 

277 ) 

278 result = await call_structured( 

279 self._primary_model, self._fallback_model, clamped_messages, self.summary_schema 

280 ) 

281 return _LoopUpdate(summary=result.parsed, summarize_cost=result.estimated_cost_usd)

283 async def assemble_node(self, state: AgentLoopState) -> TriageStateUpdate: 

284 messages = resolve_dangling_tool_calls(state["messages"]) 

285 tool_calls = derive_tool_call_records(messages) 

286 trajectory_cost = estimate_trajectory_cost(messages) 

287 summarize_cost = state.get("summarize_cost", 0.0) 

288 # Set by summarize_node with `self.summary_schema`, so this is 

289 # always either None (short-circuit path) or a SummaryT instance — 

290 # a cast, not a runtime check, since pydantic already validated it 

291 # against summary_schema when call_structured parsed it. 

292 summary = cast(SummaryT | None, state.get("summary")) 

293 

294 update = await self.finalize(summary, tool_calls, state) 

295 

296 run_meta = update.get("run_meta", state["run_meta"]) 

297 cap_hit = len(tool_calls) >= self.max_tool_calls 

298 update["run_meta"] = run_meta.with_usage( 

299 cost_usd=trajectory_cost + summarize_cost, 

300 tool_calls=len(tool_calls), 

301 iterations=1, 

302 ) 

303 log.info( 

304 "agent_subgraph_finished", 

305 node=self.name, 

306 tool_call_count=len(tool_calls), 

307 tools_used=sorted({call.tool_name for call in tool_calls}), 

308 cost_usd=trajectory_cost + summarize_cost, 

309 cap_hit=cap_hit, 

310 ) 

311 return update
