45class DrafterSubgraph(AgentSubgraph[DraftProposal]): 

46 """Turns the Planner's classification + the Researcher's findings into a 

47 concrete, grounded proposed action (or set of actions). Built exactly 

48 like `ResearcherSubgraph`: comment/label/close drafting needs no tool 

49 calls, so `tools=[]` is still the common case (e.g. `E2B_API_KEY` unset, 

50 see `tools.sandbox.sandbox_toolset`). When a run does have sandbox 

51 access, the composition root passes the sandbox's 7 tools 

52 (`tools.sandbox.build_sandbox_tools`) plus the matching `SandboxHandle` 

53 via `sandbox_handle` — the same genuine tool-calling loop 

54 (propose diff -> run sandbox -> maybe retry) as any other 

55 `AgentSubgraph`, requiring no rewiring of the parent graph. The model 

56 itself never proposes a `CodeFixAction` directly (it can't type one 

57 into existence — `ProposedAction.action` only accepts `CodeFixIntent`, 

58 intent-only); once the tool-calling loop ends, `finalize()`'s 

59 `_resolve_code_fix_intent` reads `sandbox_handle`'s already-recorded 

60 `SandboxAttempt`s and builds the real `CodeFixAction` from the last 

61 passing `fix_attempt`, degrading to a `CommentAction` describing what 

62 was tried if none passed (or no handle was configured at all). 

63 

64 `finalize()` also makes its own independent LLM call — the grounding 

65 self-check — after building the draft from `summary`. This is a 

66 genuinely separate pass from the one that produced `summary` (the 

67 evaluator-optimizer pattern the brief calls for: a model is a much 

68 weaker judge of its own claims in the same breath it wrote them), not a 

69 second field on the same structured-output call. `finalize()` only adds 

70 this call's own cost onto the `run_meta` it returns — `assemble_node` 

71 (inherited, unmodified) still owns accumulating the draft-generation 

72 cost (trajectory + summarize) and bumping `iteration_count`, exactly the 

73 same contract every `AgentSubgraph` subclass gets for free. 

74 """

114 async def finalize( 

115 self, 

116 summary: DraftProposal | None, 

117 tool_calls: list[ToolCallRecord], # noqa: ARG002 

118 state: TriageState, 

119 ) -> TriageStateUpdate: 

120 if summary is None: 

121 raise ValueError( 

122 "DrafterSubgraph.finalize() received summary=None -- prepare() never " 

123 "short-circuits, so this should be unreachable" 

124 ) 

125 research_findings = state["research_findings"] 

126 if research_findings is None: 

127 raise ValueError("DrafterSubgraph.finalize() requires research_findings to be set")

129 # `item.action` is `DraftIntent`, not yet `DraftAction` -- a 

130 # `CodeFixIntent` is resolved against `self._sandbox_handle`'s 

131 # already-recorded attempts (see `_resolve_code_fix_intent`), never 

132 # against a second one: two `CodeFixIntent`s in the same proposal 

133 # would both resolve to the same handle state, duplicating (or 

134 # misattributing) sandbox attempts across two actions, so only the 

135 # first is honored. 

136 # 

137 # `grounding_eligible_actions` tracks, in parallel, which of 

138 # `actions` are safe to send to the grounding self-check below. The 

139 # fallback-to-comment branch of `_resolve_code_fix_intent` builds its 

140 # text from `SandboxAttempt` data (files touched, test-log excerpt), 

141 # not from `ResearchFindings.evidence` -- the grounding check has no 

142 # way to verify sandbox-derived facts against evidence it was never 

143 # given, so that action is excluded from the check's input while 

144 # still being included, unmodified, in `draft.actions`. 

145 actions: list[DraftedAction] = [] 

146 grounding_eligible_actions: list[DraftedAction] = [] 

147 code_fix_resolved = False 

148 for item in summary.actions: 

149 if isinstance(item.action, CodeFixIntent): 

150 if code_fix_resolved: 

151 log.warning("duplicate_code_fix_intent_dropped") 

152 continue 

153 drafted_action, eligible_for_grounding = self._resolve_code_fix_intent(item) 

154 actions.append(drafted_action) 

155 if eligible_for_grounding: 

156 grounding_eligible_actions.append(drafted_action) 

157 code_fix_resolved = True 

158 else: 

159 drafted_action = DraftedAction(action=item.action, rationale=item.rationale) 

160 actions.append(drafted_action) 

161 grounding_eligible_actions.append(drafted_action) 

162 

163 # Sandbox spend must be folded into run_meta unconditionally -- 

164 # computed once here, before branching, so a code-fix-only draft 

165 # (which skips the grounding check below and returns early) doesn't 

166 # silently drop real E2B sandbox cost from the run's accounting. 

167 sandbox_cost = self._sandbox_handle.estimated_cost_usd if self._sandbox_handle else 0.0 

168 

169 # Only the text actually posted to GitHub is checked -- never 

170 # rationale/overall_rationale (internal reasoning, never posted, a 

171 # judgment call rather than a factual claim) -- and only from 

172 # `grounding_eligible_actions`, never `actions` wholesale, so a 

173 # degraded code-fix-fallback comment (sandbox-derived facts the 

174 # grounding check has no evidence to verify against) never reaches 

175 # it. `None` means nothing eligible produces any public-facing text 

176 # (e.g. a label-only draft, or a draft consisting solely of the 

177 # fallback comment), so there is nothing to fact-check -- skip the 

178 # grounding-check LLM call entirely rather than running it against 

179 # rationale or unverifiable sandbox facts (see 

180 # `format_public_draft_text`'s docstring for the rationale bug this 

181 # avoids). 

182 public_draft_text = format_public_draft_text(grounding_eligible_actions) 

183 if public_draft_text is None: 

184 draft = DraftOutput( 

185 actions=actions, 

186 overall_rationale=summary.overall_rationale, 

187 unsupported_claims=[], 

188 drafted_at=datetime.now(UTC), 

189 ) 

190 update = TriageStateUpdate(draft=draft, status=RunStatus.DRAFTING) 

191 # Only actually write `run_meta` when there is real sandbox 

192 # spend to fold in -- a no-op key (same value as the input) here 

193 # would blur the "total=False, absent key means unchanged" 

194 # convention this schema otherwise follows (see 

195 # `ResearcherSubgraph`'s own short-circuit path). 

196 if sandbox_cost: 

197 update["run_meta"] = state["run_meta"].with_usage(cost_usd=sandbox_cost) 

198 return update

200 # Independent second LLM call: the grounding self-check. See class 

201 # docstring for why this must be a genuinely separate pass rather 

202 # than a second field on the call that produced `summary`. 

203 critique_messages = GROUNDING_CHECK_PROMPT.format_messages( 

204 draft_text=public_draft_text, 

205 evidence=format_evidence_for_prompt(research_findings.evidence), 

206 ) 

207 critique_result = await call_structured( 

208 self._primary_model, self._fallback_model, critique_messages, GroundingCritique 

209 ) 

210 

211 draft = DraftOutput( 

212 actions=actions, 

213 overall_rationale=summary.overall_rationale, 

214 unsupported_claims=critique_result.parsed.unsupported_claims, 

215 drafted_at=datetime.now(UTC), 

216 ) 

217 # Only this call's own cost (plus this run's sandbox spend, if any) is 

218 # added here -- never replicate assemble_node's trajectory/summarize 

219 # accumulation or touch iteration_count; assemble_node adds the 

220 # draft-generation cost and the iteration bump on top of whatever 

221 # run_meta this method returns. 

222 updated_run_meta = state["run_meta"].with_usage( 

223 cost_usd=critique_result.estimated_cost_usd + sandbox_cost 

224 ) 

225 return TriageStateUpdate(draft=draft, status=RunStatus.DRAFTING, run_meta=updated_run_meta)

227 def _resolve_code_fix_intent(self, item: ProposedAction) -> tuple[DraftedAction, bool]: 

228 """Turns a model-proposed `CodeFixIntent` into a real `DraftedAction`, 

229 reading only `self._sandbox_handle`'s already-recorded state -- never 

230 touching the sandbox itself (no `ensure_ready`/tool call happens 

231 here; that's the tool-calling loop's job, already finished by the 

232 time `finalize()` runs). 

233 

234 Returns `(drafted_action, eligible_for_grounding_check)`. The 

235 fallback-to-comment branch below builds its text from 

236 `SandboxAttempt` data (files touched, a test-log excerpt) rather 

237 than `ResearchFindings.evidence`, so it comes back with 

238 `eligible_for_grounding_check=False` -- the grounding self-check has 

239 no way to verify sandbox-derived facts against evidence it was never 

240 given. The action is still returned (and still lands in 

241 `draft.actions`) either way; only its eligibility for the grounding 

242 check's input differs.""" 

243 handle = self._sandbox_handle 

244 attempt = handle.last_passing_fix_attempt if handle is not None else None 

245 if handle is None or attempt is None: 

246 # No verified fix exists: degrade to a comment describing the 

247 # attempt rather than silently omitting it (DraftOutput.actions 

248 # requires min_length=1, and a human reviewer benefits from 

249 # visibility into what was tried and why it didn't pass). 

250 return ( 

251 DraftedAction( 

252 action=CommentAction( 

253 comment_body=format_failed_fix_comment( 

254 handle.attempts if handle else [], 

255 install_attempted=handle.install_attempted if handle else False, 

256 ) 

257 ), 

258 rationale=item.rationale, 

259 ), 

260 False, 

261 ) 

262 if handle.base_commit_sha is None or handle.base_ref is None: 

263 # Invariant: `run_tests` is the only method that appends to 

264 # `attempts`, and it always calls `_ensure_ready_locked()` first 

265 # -- which sets both fields before any attempt can exist. A 

266 # passing `attempt` here means the sandbox was already set up, 

267 # so this should be unreachable outside a `SandboxHandle` bug. 

268 raise ValueError( 

269 "SandboxHandle has a passing fix attempt but no base_commit_sha/base_ref recorded" 

270 ) 

271 return ( 

272 DraftedAction( 

273 action=CodeFixAction( 

274 diff=attempt.diff, 

275 target_files=attempt.changed_files, 

276 sandbox_result=attempt.result, 

277 base_commit_sha=handle.base_commit_sha, 

278 base_ref=handle.base_ref, 

279 ), 

280 rationale=item.rationale, 

281 ), 

282 # `public_facing_text` returns `None` for `code_fix` actions 

283 # regardless (a passing diff is never posted verbatim as 

284 # GitHub-comment prose), so this eligibility bit is inert today 

285 # -- marked `True` for correctness/symmetry, since a passing 

286 # `CodeFixAction`'s `sandbox_result` genuinely is the kind of 

287 # system-derived fact this check exists to trust, not flag. 

288 True, 

289 )
