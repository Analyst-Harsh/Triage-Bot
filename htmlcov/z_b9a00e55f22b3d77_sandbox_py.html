1"""E2B sandbox tool layer for the Drafter's code-fix loop. 

2 

3Owns the lifecycle of one ephemeral E2B sandbox per Drafter run: fetching the 

4target repo at a resolved commit, gating network egress through three phases 

5(GitHub-tarball-hosts-only -> package-registries-only -> locked), running the 

6agent's install/edit/test tool calls against it, and recording every 

7(diff, test-result) attempt as a `SandboxAttempt` for `DrafterSubgraph` to 

8consume. Mirrors `tools/mcp_clients.py`'s composition-root pattern 

9(`sandbox_toolset` == `researcher_toolset`) but diverges on eagerness: no E2B 

10or GitHub I/O happens until a tool is actually called (see `ensure_ready`). 

11"""

47# Unlike fix_attempt, baseline/repro have no natural cap on "productive" 

48# retries (a model can legitimately need a couple of tries to find the right 

49# invocation) -- but with no cap at all, a baseline that's failing for a 

50# structural reason (a missing test dependency, not a bad test-command 

51# invocation) can absorb the entire tool-call budget on cosmetic invocation 

52# variants that were never going to work, leaving zero calls for the fix 

53# itself. These caps exist to force the model back to install_dependencies 

54# (via the ERROR text below) well before that happens. 

55MAX_SANDBOX_BASELINE_ATTEMPTS = 3 

56MAX_SANDBOX_REPRO_ATTEMPTS = 3

59# configurable: it's an implementation detail of this module, never exposed 

60# to the agent (tool paths are relative and resolved against it internally).

94def _tail_clamp(text: str, max_chars: int) -> str: 

95 """Like `clamp_tool_output`'s clamp, but keeps the *tail* of `text` 

96 rather than the head. Test-runner and installer output puts the useful 

97 part -- the FAILURES/traceback section, the final pass/fail summary 

98 line, pip/npm's error block -- at the end of stdout+stderr, not the 

99 start, so head-cropping it (as the generic byte-oriented 

100 `clamp_tool_output` does) throws away exactly the part a caller needs to 

101 diagnose a failure.""" 

102 if len(text) <= max_chars: 

103 return text 

104 omitted = len(text) - max_chars 

105 return f"...[truncated, {omitted} earlier characters]\n{text[-max_chars:]}"

127class SandboxHandle: 

128 """Owns the lazy E2B sandbox for one Drafter run. 

129 

130 Every public method acquires `self._lock` (an `asyncio.Lock`) for its 

131 whole body, including reads -- the sandbox is a single sequential 

132 resource, and this avoids a parallel tool-call batch racing a write 

133 against a diff snapshot. `ensure_ready()` acquires the same lock itself 

134 and delegates to `_ensure_ready_locked()`; every other method calls 

135 `_ensure_ready_locked()` directly (never the public `ensure_ready()`) 

136 since it's already holding the lock by the time it gets there -- 

137 `asyncio.Lock` isn't reentrant, so re-acquiring it from inside itself 

138 would deadlock. 

139 

140 Contract shared by every public tool method (read_file/write_file/ 

141 edit_file/list_files/install_dependencies/run_tests): never raises. 

142 `SandboxSetupError` and any E2B SDK exception are caught internally, 

143 logged as a structured warning (`sandbox_tool_error`: tool, error), and 

144 turned into a returned "ERROR: ..." string. 

145 

146 `repo_full_name`/`ref` are per-run values, supplied by the composition 

147 root from `state["issue"].repo_full_name` -- never fixed at graph-build 

148 time. 

149 """ 

150

180 async def _resolve_ref_and_tarball_url(self) -> tuple[str, str, str]: 

181 """Resolves `self._ref` (or the repo's default branch, if unset) to 

182 a real commit SHA and a signed tarball download URL, via PyGithub. 

183 

184 PyGithub is a synchronous/blocking client (built on `requests`), so 

185 this whole round trip runs off the event loop via `asyncio.to_thread` 

186 rather than blocking it directly inside an async method. 

187 """ 

188 

189 def _resolve() -> tuple[str, str, str]: 

190 repo = self._github_client.get_repo(self._repo_full_name) 

191 resolved_ref = self._ref if self._ref is not None else repo.default_branch 

192 sha = repo.get_commit(resolved_ref).sha 

193 tarball_url = repo.get_archive_link("tarball", sha) 

194 return resolved_ref, sha, tarball_url 

195 

196 return await asyncio.to_thread(_resolve)

198 async def _ensure_ready_locked(self) -> None: 

199 """Create-on-first-use. Assumes `self._lock` is already held by the 

200 caller. Idempotent -- only does real work once.""" 

201 if self._sandbox is not None: 

202 return 

203 if self._settings.e2b_api_key is None: 

204 raise SandboxSetupError("E2B_API_KEY is not configured") 

205 

206 sandbox: AsyncSandbox | None = None 

207 try: 

208 # Phase 1: GitHub hosts only, so this sandbox can do nothing but 

209 # fetch the repo before any package installs or agent content 

210 # exist. Restriction is skipped entirely when 

211 # e2b_restrict_network is False (local-debugging escape hatch); 

212 # the network is still locked unconditionally later, in 

213 # _lock_network -- that boundary protects against agent-authored 

214 # content phoning home and isn't part of the debugging opt-out. 

215 network: SandboxNetworkOpts | None = None 

216 if self._settings.e2b_restrict_network: 

217 network = SandboxNetworkOpts( 

218 allow_out=list(_GITHUB_TARBALL_HOSTS), deny_out=[ALL_TRAFFIC] 

219 ) 

220 

221 sandbox = await AsyncSandbox.create( 

222 timeout=int(self._settings.e2b_sandbox_session_timeout_seconds), 

223 api_key=self._settings.e2b_api_key.get_secret_value(), 

224 network=network, 

225 ) 

226 

227 resolved_ref, sha, tarball_url = await self._resolve_ref_and_tarball_url() 

228 

229 setup_timeout = int(self._settings.e2b_install_timeout_seconds) 

230 # Never `git clone` -- fetch+extract the tarball at the resolved 

231 # SHA instead, then synthesize a local git history so `git diff` 

232 # has a baseline (GitHub tarballs ship with no .git). 

233 await sandbox.commands.run(f"mkdir -p {_REPO_DIR}", timeout=setup_timeout) 

234 await sandbox.commands.run( 

235 f'curl -fsSL "{tarball_url}" | tar -xz -C {_REPO_DIR} --strip-components=1', 

236 timeout=setup_timeout, 

237 ) 

238 await sandbox.commands.run( 

239 "git init -q && git add -A && " 

240 "git -c user.email=sandbox@triage-bot.local -c user.name=triage-bot " 

241 "commit -q -m 'sandbox baseline' --allow-empty", 

242 cwd=_REPO_DIR, 

243 timeout=setup_timeout, 

244 ) 

245 

246 # Phase 2: package registries only, so install_dependencies can 

247 # run next but nothing else can reach the network. 

248 if self._settings.e2b_restrict_network: 

249 await sandbox.update_network( 

250 SandboxNetworkUpdate( 

251 allow_out=list(_PACKAGE_REGISTRY_HOSTS), deny_out=[ALL_TRAFFIC] 

252 ) 

253 ) 

254 

255 self._sandbox = sandbox 

256 # The *real* GitHub SHA the tarball was fetched at -- NOT the 

257 # synthetic local commit's SHA from the `git init` step above. 

258 self.base_commit_sha = sha 

259 self.base_ref = resolved_ref 

260 except Exception as exc: 

261 # Don't leak a sandbox that was created but failed a later setup 

262 # step (tarball fetch, git init, network transition). 

263 if sandbox is not None and self._sandbox is None: 

264 with contextlib.suppress(Exception): 

265 await sandbox.kill() 

266 log.warning( 

267 "sandbox_setup_failed", 

268 repo=self._repo_full_name, 

269 ref=self._ref, 

270 error=str(exc), 

271 ) 

272 raise SandboxSetupError( 

273 f"failed to set up sandbox for {self._repo_full_name}@{self._ref}: {exc}" 

274 ) from exc

291 async def _run_command( 

292 self, command: str, *, timeout_seconds: int, cwd: str | None = None 

293 ) -> CommandResult: 

294 """Runs `command` and always returns a `CommandResult`, even on a 

295 non-zero exit code -- unlike the E2B SDK's default of raising 

296 `CommandExitException` in that case. A failing install or test 

297 command is an expected, meaningful outcome here (e.g. "tests 

298 failed"), not a sandbox-level error; genuine SDK/transport 

299 exceptions still propagate to the caller unchanged. 

300 

301 Named `timeout_seconds`, not `timeout` -- ruff's ASYNC109 flags an 

302 async function with a bare `timeout` parameter as if it were 

303 implementing its own cancellation logic (suggesting 

304 `asyncio.timeout` instead); this just forwards the value to the E2B 

305 SDK's own `commands.run(timeout=...)`, so that rule doesn't apply.""" 

306 sandbox = self._require_sandbox() 

307 try: 

308 return await sandbox.commands.run(command, timeout=timeout_seconds, cwd=cwd) 

309 except CommandExitException as exc: 

310 return CommandResult( 

311 stdout=exc.stdout, stderr=exc.stderr, exit_code=exc.exit_code, error=exc.error 

312 )

320 def _bounded_timeout_seconds(self, configured_timeout: float) -> int: 

321 """Caps a per-call timeout at whatever's left of the run's 

322 billed-seconds budget, so a single slow command can no longer run 

323 for the full `configured_timeout` once the budget is nearly 

324 exhausted -- without this, a call starting with e.g. 1s of 

325 remaining budget could still run for the full 

326 e2b_install_timeout_seconds (300s) before the next gate check 

327 fires. Floors at 1 second (never 0, which some execution backends 

328 treat as "no timeout" rather than "immediate timeout") -- every 

329 caller already refuses the call outright when `self._billed_seconds 

330 >= e2b_max_billed_seconds_per_run`, so this is only reached with a 

331 strictly positive remaining budget, potentially under a second. 

332 """ 

333 remaining_budget = self._settings.e2b_max_billed_seconds_per_run - self._billed_seconds 

334 return max(1, int(min(configured_timeout, remaining_budget)))

336 @staticmethod 

337 def _repeat_guarded( 

338 fn: Callable[Concatenate[SandboxHandle, _P], Awaitable[str]], 

339 ) -> Callable[Concatenate[SandboxHandle, _P], Awaitable[str]]: 

340 """Decorator for the read-only tool methods (list_files/read_file/ 

341 search_file). Refuses an exact repeat of the call immediately before 

342 it -- these tools are deterministic (their own descriptions already 

343 say so), so an immediate repeat can't surface anything new and only 

344 wastes a tool-call slot out of the Drafter's budget (the failure 

345 mode this closes: a model looping on the same read_file range 

346 indefinitely). Binds arguments against the function's own signature 

347 (not the raw positional/keyword args a caller happened to use) so 

348 the same logical call always produces the same key regardless of 

349 calling style. 

350 

351 Compares against a single most-recent-call slot (`self._last_call`), 

352 not a running history of every call made this run: a full-history 

353 blacklist would permanently refuse a *later*, legitimate re-fetch of 

354 the same arguments -- e.g. once ContextEditingMiddleware clears an 

355 old tool result, its placeholder text explicitly tells the model to 

356 "call this tool again if you need the original data" 

357 (see `graph/nodes/agent_subgraph.py`'s `context_edit_placeholder`). 

358 `self._last_call` is shared with every other public tool method -- 

359 write_file/edit_file/install_dependencies/run_tests each reset it to 

360 `None` at the top of their own body (undecorated) -- so adjacency is 

361 measured against the true previous tool call, not just the most 

362 recent read-only one (e.g. read_file -> edit_file -> the same 

363 read_file again must NOT be flagged as a repeat). 

364 

365 The check-and-record step is done under `self._lock` (briefly 

366 acquired and released, not held across the call into `fn`) to keep 

367 it race-free against a concurrent tool-call batch, consistent with 

368 this class's existing concurrency contract -- then released before 

369 calling `fn`, which acquires the same (non-reentrant) lock itself 

370 for its own body. 

371 """ 

372 signature = inspect.signature(fn) 

373 

374 @functools.wraps(fn) 

375 async def wrapper(self: SandboxHandle, *args: _P.args, **kwargs: _P.kwargs) -> str: 

376 bound = signature.bind(self, *args, **kwargs) 

377 bound.apply_defaults() 

378 call_args = dict(bound.arguments) 

379 del call_args["self"] 

380 key = (fn.__name__, tuple(call_args.items())) 

381 async with self._lock: 

382 is_repeat = self._last_call == key 

383 self._last_call = key 

384 if is_repeat: 

385 log.info("sandbox_repeat_call_refused", tool=fn.__name__, **call_args) 

386 return ( 

387 f"ERROR: {fn.__name__} was just called with these exact " 

388 "arguments -- repeating it immediately can't reveal anything " 

389 "new (these tools are deterministic). Change the arguments, " 

390 "or move on to the next step, instead of repeating this " 

391 "exact call." 

392 ) 

393 return await fn(self, *args, **kwargs) 

394 

395 return wrapper

397 @_repeat_guarded 

398 async def read_file( 

399 self, path: str, start_line: int | None = None, end_line: int | None = None 

400 ) -> str: 

401 """`start_line`/`end_line` (1-indexed, both optional) let the model 

402 read a specific window of a large file instead of always the head -- 

403 without them, `arrow/arrow.py`-sized files (65KB, well over the 

404 16,000-char default clamp) can never have their tail seen at all, 

405 no matter how many times read_file is called. Self-clamps via 

406 `self._settings.drafter_file_read_max_chars`, the same pattern 

407 `install_dependencies`/`run_tests` already use for their own output 

408 budgets -- so `build_sandbox_tools` no longer needs to wrap this 

409 tool in the generic `clamp_tool_output`.""" 

410 log.info("sandbox_read_file", path=path, start_line=start_line, end_line=end_line) 

411 async with self._lock: 

412 try: 

413 await self._ensure_ready_locked() 

414 sandbox = self._require_sandbox() 

415 text = await sandbox.files.read(self._resolve_path(path)) 

416 return self._slice_and_clamp_lines(text, start_line=start_line, end_line=end_line) 

417 except Exception as exc: 

418 log.warning("sandbox_tool_error", tool="read_file", error=str(exc)) 

419 return f"ERROR: {exc}{await self._list_parent_for_error(path)}" 

420 

421 def _slice_and_clamp_lines( 

422 self, text: str, *, start_line: int | None, end_line: int | None 

423 ) -> str: 

424 """Shared by `read_file`'s whole-file default and explicit-range 

425 requests -- always reports total line count in its truncation 

426 notice (never just an opaque char count) so the model knows how far 

427 it can page, per the "tool calls are deterministic" Global 

428 Constraint in `prompts/drafter.py`: repeating the same call can 

429 never reveal more, but a different `start_line`/`end_line` can.""" 

430 lines = text.splitlines() 

431 total_lines = len(lines) 

432 

433 if start_line is None and end_line is None: 

434 first_line, last_line = 1, total_lines 

435 else: 

436 first_line = start_line if start_line is not None else 1 

437 if first_line < 1: 

438 return "ERROR: start_line must be >= 1" 

439 if first_line > total_lines: 

440 return f"ERROR: file has only {total_lines} lines" 

441 last_line = ( 

442 end_line if end_line is not None else first_line + _DEFAULT_READ_WINDOW_LINES - 1 

443 ) 

444 if last_line < first_line: 

445 return "ERROR: end_line must be >= start_line" 

446 last_line = min(last_line, total_lines) 

447 

448 content = "\n".join(lines[first_line - 1 : last_line]) 

449 

450 max_chars = self._settings.drafter_file_read_max_chars 

451 char_truncated = len(content) > max_chars 

452 if char_truncated: 

453 content = content[:max_chars] 

454 

455 if last_line < total_lines or char_truncated: 

456 content += ( 

457 f"\n...[truncated: showing lines {first_line}-{last_line} of " 

458 f"{total_lines} total. Use start_line/end_line to read further, or " 

459 "search_file to jump to something specific.]" 

460 ) 

461 return content

463 @_repeat_guarded 

464 async def search_file(self, path: str, pattern: str, context_lines: int = 3) -> str: 

465 """Literal substring search, deliberately not regex -- a 

466 model-chosen (or content-influenced, via prompt injection from file 

467 contents) regex pattern would be a ReDoS vector for no real benefit 

468 over substring matching when looking for a function/snippet name. 

469 Complements `read_file`'s line range for the case evidence gives no 

470 line number at all (`Evidence.reference` is a free-form string per 

471 its own field description, not guaranteed to carry one).""" 

472 log.info("sandbox_search_file", path=path, pattern=pattern) 

473 async with self._lock: 

474 try: 

475 if not pattern: 

476 return "ERROR: pattern must not be empty" 

477 await self._ensure_ready_locked() 

478 sandbox = self._require_sandbox() 

479 text = await sandbox.files.read(self._resolve_path(path)) 

480 return self._format_search_result( 

481 text, path=path, pattern=pattern, context_lines=context_lines 

482 ) 

483 except Exception as exc: 

484 log.warning("sandbox_tool_error", tool="search_file", error=str(exc)) 

485 return f"ERROR: {exc}{await self._list_parent_for_error(path)}" 

486 

487 def _format_search_result( 

488 self, text: str, *, path: str, pattern: str, context_lines: int 

489 ) -> str: 

490 lines = text.splitlines() 

491 matches = [i for i, line in enumerate(lines) if pattern in line] 

492 if not matches: 

493 return f"No matches for {pattern!r} in {path}" 

494 

495 shown = matches[:_MAX_SEARCH_MATCHES] 

496 blocks: list[str] = [] 

497 for idx in shown: 

498 start = max(0, idx - context_lines) 

499 end = min(len(lines), idx + context_lines + 1) 

500 blocks.append("\n".join(f"line {i + 1}: {lines[i]}" for i in range(start, end))) 

501 content = "\n--\n".join(blocks) 

502 

503 if len(matches) > _MAX_SEARCH_MATCHES: 

504 omitted_matches = len(matches) - _MAX_SEARCH_MATCHES 

505 content += f"\n...[{omitted_matches} further matches omitted, narrow your pattern]" 

506 

507 max_chars = self._settings.drafter_file_read_max_chars 

508 if len(content) > max_chars: 

509 omitted = len(content) - max_chars 

510 content = f"{content[:max_chars]}\n...[truncated, {omitted} more characters]" 

511 return content

513 async def _list_parent_for_error(self, path: str) -> str: 

514 """Best-effort: appended to a failed `read_file`'s error so the model 

515 can self-correct (find the real path) without spending a separate 

516 `list_files` round trip on a path it already knows is wrong. Never 

517 raises -- any failure here just falls back to the plain error 

518 string. Calls `sandbox.files.list()` directly rather than the public 

519 `list_files()` method: the caller (`read_file`) already holds 

520 `self._lock`, and `asyncio.Lock` isn't reentrant.""" 

521 log.info("sandbox_list_parent_for_error", path=path) 

522 try: 

523 sandbox = self._require_sandbox() 

524 parent = self._resolve_path(path).rsplit("/", 1)[0] or "/" 

525 entries = await sandbox.files.list(parent) 

526 names = "\n".join( 

527 f"{entry.path}/" if entry.type == FileType.DIR else entry.path for entry in entries 

528 ) 

529 return f"\nContents of {parent}:\n{names}" if names else f"\n{parent} is empty" 

530 except Exception: 

531 return ""

533 async def write_file(self, path: str, content: str) -> str: 

534 log.info("sandbox_write_file", path=path, content=content) 

535 async with self._lock: 

536 try: 

537 # Invalidates the read-only tools' repeat-guard adjacency 

538 # (see _repeat_guarded): a mutation happened, so a 

539 # subsequent read_file/search_file/list_files call that 

540 # repeats an earlier one is no longer a back-to-back 

541 # no-op and must not be refused. 

542 self._last_call = None 

543 await self._ensure_ready_locked() 

544 sandbox = self._require_sandbox() 

545 await self._lock_network() 

546 # e2b's Filesystem.write signature carries an unparameterized 

547 # `IO[Unknown]` in its `data` union, which makes the awaited 

548 # return type only partially known to pyright regardless of 

549 # the (fully-str) arguments actually passed here -- a stub 

550 # completeness gap in the third-party library, not a type 

551 # error in this call. 

552 await sandbox.files.write( # pyright: ignore[reportUnknownMemberType] 

553 self._resolve_path(path), content 

554 ) 

555 return f"wrote {path}" 

556 except Exception as exc: 

557 log.warning("sandbox_tool_error", tool="write_file", error=str(exc)) 

558 return f"ERROR: {exc}"

560 async def edit_file(self, path: str, find: str, replace: str) -> str: 

561 log.info("sandbox_edit_file", path=path, find=find, replace=replace) 

562 async with self._lock: 

563 try: 

564 # See the comment in write_file() above. 

565 self._last_call = None 

566 await self._ensure_ready_locked() 

567 sandbox = self._require_sandbox() 

568 await self._lock_network() 

569 resolved_path = self._resolve_path(path) 

570 text = await sandbox.files.read(resolved_path) 

571 occurrences = text.count(find) 

572 if occurrences == 0: 

573 return f"ERROR: no match for the given text in {path}" 

574 if occurrences > 1: 

575 return ( 

576 f"ERROR: {occurrences} matches for the given text in {path}; " 

577 "must match exactly once" 

578 ) 

579 # See the comment on the write() call in write_file() above. 

580 await sandbox.files.write( # pyright: ignore[reportUnknownMemberType] 

581 resolved_path, text.replace(find, replace, 1) 

582 ) 

583 return f"edited {path}" 

584 except Exception as exc: 

585 log.warning("sandbox_tool_error", tool="edit_file", error=str(exc)) 

586 return f"ERROR: {exc}"

606 async def install_dependencies(self, command: str) -> str: 

607 async with self._lock: 

608 try: 

609 # See the comment in write_file() above. 

610 self._last_call = None 

611 await self._ensure_ready_locked() 

612 self._require_sandbox() 

613 if self._network_locked: 

614 log.info( 

615 "sandbox_tool_result", 

616 tool="install_dependencies", 

617 command=command, 

618 passed=False, 

619 reason="network_locked", 

620 ) 

621 return ( 

622 "ERROR: sandbox network is locked; dependencies can no longer be installed" 

623 ) 

624 if not _is_recognized_installer(command): 

625 log.info( 

626 "sandbox_tool_result", 

627 tool="install_dependencies", 

628 command=command, 

629 passed=False, 

630 reason="unrecognized_installer", 

631 ) 

632 return f"ERROR: unrecognized installer command: {command!r}" 

633 if self._billed_seconds >= self._settings.e2b_max_billed_seconds_per_run: 

634 log.info( 

635 "sandbox_tool_result", 

636 tool="install_dependencies", 

637 command=command, 

638 passed=False, 

639 reason="billed_seconds_exhausted", 

640 ) 

641 return "ERROR: sandbox billed-seconds budget exhausted" 

642 

643 # Marked as soon as a real install command is genuinely 

644 # dispatched -- regardless of its exit code -- so run_tests 

645 # can tell "the agent tried and it's still failing" apart 

646 # from "the agent skipped straight to run_tests" (see 

647 # run_tests's own baseline-without-install gate below). 

648 self._install_attempted = True 

649 

650 start = time.monotonic() 

651 result = await self._run_command( 

652 command, 

653 timeout_seconds=self._bounded_timeout_seconds( 

654 self._settings.e2b_install_timeout_seconds 

655 ), 

656 cwd=_REPO_DIR, 

657 ) 

658 combined = f"{result.stdout}{result.stderr}" 

659 

660 self._billed_seconds += time.monotonic() - start 

661 

662 passed = result.exit_code == 0 

663 status = "INSTALLED" if passed else "INSTALL_FAILED" 

664 budget = ( 

665 self._settings.drafter_test_log_success_max_chars 

666 if passed 

667 else self._settings.drafter_test_log_failure_max_chars 

668 ) 

669 log.info( 

670 "sandbox_tool_result", 

671 tool="install_dependencies", 

672 command=command, 

673 exit_code=result.exit_code, 

674 # result=combined, 

675 duration_seconds=time.monotonic() - start, 

676 passed=passed, 

677 ) 

678 return f"{status}: {command}\n{_tail_clamp(combined, budget)}" 

679 except Exception as exc: 

680 log.warning("sandbox_tool_error", tool="install_dependencies", error=str(exc)) 

681 return f"ERROR: {exc}"

683 async def run_tests( 

684 self, *, kind: Literal["baseline", "repro", "fix_attempt"], test_command: str 

685 ) -> str: 

686 async with self._lock: 

687 try: 

688 # See the comment in write_file() above. 

689 self._last_call = None 

690 await self._ensure_ready_locked() 

691 self._require_sandbox() 

692 # A "baseline" run exercises only the pristine, as-fetched 

693 # repo -- no agent-authored content exists in the sandbox 

694 # yet (write_file/edit_file always lock the network 

695 # unconditionally, before writing anything, so if either had 

696 # already run this call would find the network already 

697 # locked regardless of `kind`). That makes it safe to run 

698 # baseline during the install-phase (registry-only) network, 

699 # which some test runners need: tox/nox provision their own 

700 # per-env dependencies as part of the *first* invocation of a 

701 # given environment, rather than during a separate install 

702 # step, and would otherwise fail to reach the registry once 

703 # the network is locked. Every other kind runs strictly 

704 # after an edit has already locked the network -- this call 

705 # is a no-op in that case, not a widened window. 

706 if kind != "baseline": 

707 await self._lock_network() 

708 

709 # A first baseline attempt is always allowed without a prior 

710 # install_dependencies call (see the network-gating comment 

711 # above -- tox/nox self-provision on first invocation). But 

712 # once that first attempt has failed with nothing installed, 

713 # a further baseline attempt is refused outright rather than 

714 # just hinted at: a hint is something the model can read and 

715 # ignore on retry -- exactly the failure mode this gate 

716 # exists to close. This never fires once any baseline 

717 # attempt has passed, so a tox/nox repo that already went 

718 # green is never blocked from anything. 

719 def _refuse(reason: str, message: str) -> str: 

720 log.info( 

721 "sandbox_tool_result", 

722 tool="run_tests", 

723 kind=kind, 

724 test_command=test_command, 

725 passed=False, 

726 reason=reason, 

727 ) 

728 return message 

729 

730 if ( 

731 kind == "baseline" 

732 and not self._has_passing_baseline() 

733 and self._attempt_count("baseline") >= 1 

734 and not self._install_attempted 

735 ): 

736 return _refuse( 

737 "baseline_retry_without_install", 

738 "ERROR: the previous baseline attempt failed and " 

739 "install_dependencies has not been called yet this run -- call " 

740 "install_dependencies before retrying run_tests, rather than " 

741 "repeating the same test command.", 

742 ) 

743 if ( 

744 kind == "baseline" 

745 and self._attempt_count("baseline") >= MAX_SANDBOX_BASELINE_ATTEMPTS 

746 ): 

747 return _refuse( 

748 "baseline_attempt_cap", 

749 f"ERROR: baseline attempt limit ({MAX_SANDBOX_BASELINE_ATTEMPTS}) reached.", 

750 ) 

751 if kind in ("repro", "fix_attempt") and not self._has_passing_baseline(): 

752 return _refuse( 

753 "no_passing_baseline", 'ERROR: no passing "baseline" run recorded yet' 

754 ) 

755 if kind == "repro" and self._attempt_count("repro") >= MAX_SANDBOX_REPRO_ATTEMPTS: 

756 return _refuse( 

757 "repro_attempt_cap", 

758 f"ERROR: repro attempt limit ({MAX_SANDBOX_REPRO_ATTEMPTS}) reached", 

759 ) 

760 if kind == "fix_attempt" and self._attempt_count("repro") == 0: 

761 return _refuse( 

762 "fix_attempt_without_repro", 

763 'ERROR: no "repro" run recorded yet -- run_tests(kind="repro") against ' 

764 "a test that captures this issue (write one first if none exists) " 

765 "before attempting a fix, so there is something concrete to verify the " 

766 "fix against.", 

767 ) 

768 if ( 

769 kind == "fix_attempt" 

770 and self._attempt_count("fix_attempt") >= MAX_SANDBOX_FIX_ATTEMPTS 

771 ): 

772 return _refuse( 

773 "fix_attempt_cap", 

774 f"ERROR: fix attempt limit ({MAX_SANDBOX_FIX_ATTEMPTS}) reached", 

775 ) 

776 if self._billed_seconds >= self._settings.e2b_max_billed_seconds_per_run: 

777 return _refuse( 

778 "billed_seconds_exhausted", "ERROR: sandbox billed-seconds budget exhausted" 

779 ) 

780 

781 # Snapshot the diff *before* running the test command so a 

782 # pass/fail is never separated from the exact diff that 

783 # produced it. Stage everything first: plain `git diff` 

784 # compares the working tree against the index and never 

785 # shows untracked files at all, so a brand-new file the 

786 # agent just wrote (write_file) would otherwise be 

787 # permanently invisible here -- `git add -A` (respecting 

788 # .gitignore) plus `--cached` diffs against the one-time 

789 # baseline commit from ensure_ready(), covering new files 

790 # exactly like modified ones. 

791 await self._run_command("git add -A", timeout_seconds=60, cwd=_REPO_DIR) 

792 diff_result = await self._run_command( 

793 "git diff --cached", timeout_seconds=60, cwd=_REPO_DIR 

794 ) 

795 changed_files_result = await self._run_command( 

796 "git diff --cached --name-only", timeout_seconds=60, cwd=_REPO_DIR 

797 ) 

798 changed_files = [line for line in changed_files_result.stdout.splitlines() if line] 

799 

800 # A fix_attempt whose diff exactly matches one that already 

801 # passed can't produce a different result -- refusing it here 

802 # (before the real test command runs) is what actually saves 

803 # the E2B billed-seconds a repeat call would otherwise waste, 

804 # not just a wasted tool-call slot. Deliberately compares 

805 # against the diff, not just "any passing fix_attempt exists": 

806 # a genuinely new edit made after an earlier pass must still 

807 # be verified. 

808 if ( 

809 kind == "fix_attempt" 

810 and self.last_passing_fix_attempt is not None 

811 and diff_result.stdout == self.last_passing_fix_attempt.diff 

812 ): 

813 return _refuse( 

814 "fix_attempt_diff_already_passed", 

815 "ERROR: a fix_attempt with this exact diff already passed " 

816 f"(attempt {self.last_passing_fix_attempt.attempt_number}) -- " 

817 "re-running an unchanged diff cannot produce a different result. " 

818 "Propose the code_fix action now instead of calling run_tests again.", 

819 ) 

820 

821 start = time.monotonic() 

822 test_result = await self._run_command( 

823 test_command, 

824 timeout_seconds=self._bounded_timeout_seconds( 

825 self._settings.e2b_test_command_timeout_seconds 

826 ), 

827 cwd=_REPO_DIR, 

828 ) 

829 duration = time.monotonic() - start 

830 self._billed_seconds += duration 

831 

832 passed = test_result.exit_code == 0 

833 logs = f"{test_result.stdout}{test_result.stderr}" 

834 attempt = SandboxAttempt( 

835 kind=kind, 

836 attempt_number=len(self.attempts) + 1, 

837 diff=diff_result.stdout, 

838 changed_files=changed_files, 

839 result=SandboxResult( 

840 passed=passed, 

841 logs=logs, 

842 test_command=test_command, 

843 duration_seconds=duration, 

844 ), 

845 recorded_at=datetime.now(UTC), 

846 ) 

847 self.attempts.append(attempt) 

848 log.info( 

849 "sandbox_attempt_recorded", 

850 kind=kind, 

851 test_command=test_command, 

852 attempt_number=attempt.attempt_number, 

853 changed_files=changed_files, 

854 diff_length=len(attempt.diff), 

855 logs_length=len(logs), 

856 # logs=logs, 

857 duration_seconds=duration, 

858 passed=passed, 

859 cumulative_billed_seconds=self._billed_seconds, 

860 cumulative_cost_usd=self.estimated_cost_usd, 

861 ) 

862 status = "PASSED" if passed else "FAILED" 

863 budget = ( 

864 self._settings.drafter_test_log_success_max_chars 

865 if passed 

866 else self._settings.drafter_test_log_failure_max_chars 

867 ) 

868 result_text = f"{status}: {test_command}\n{_tail_clamp(logs, budget)}" 

869 if kind == "baseline" and not passed and not self._install_attempted: 

870 # First failure, install never attempted: warn now, on 

871 # this attempt -- the *next* baseline attempt without an 

872 # install call in between is hard-refused (see the gate 

873 # above), so this is the model's one chance to notice 

874 # before that happens. 

875 result_text += ( 

876 "\n\nNote: install_dependencies has not been called yet this " 

877 "run. If this failure looks like a missing dependency (e.g. a " 

878 "module/import/command-not-found error), call " 

879 "install_dependencies before retrying." 

880 ) 

881 if kind == "repro" and passed: 

882 # Advisory, not a hard refusal -- unlike the fix_attempt 

883 # gate above, a passing repro isn't always wrong (the 

884 # issue may already be fixed), so this shouldn't block 

885 # the workflow, only prompt the model to reconsider. 

886 result_text += ( 

887 "\n\nWarning: this repro run PASSED. Per the workflow, a repro " 

888 "is expected to FAIL, capturing the bug -- a passing repro means " 

889 "the bug wasn't actually reproduced. Reconsider before editing: " 

890 "find or write a test that actually fails first." 

891 ) 

892 return result_text 

893 except Exception as exc: 

894 log.warning("sandbox_tool_error", tool="run_tests", error=str(exc)) 

895 return f"ERROR: {exc}"

898 def install_attempted(self) -> bool: 

899 """Whether `install_dependencies` has actually dispatched a real 

900 install command at least once this run, regardless of pass/fail. 

901 Used by `run_tests`'s own baseline-without-install gate, and 

902 threaded into `format_failed_fix_comment` so a "no fix attempted" 

903 comment can tell a genuine pre-existing repo failure apart from the 

904 agent's own sandbox-setup miss.""" 

905 return self._install_attempted

907 @property 

908 def last_passing_fix_attempt(self) -> SandboxAttempt | None: 

909 """Scans `self.attempts` in reverse for the last entry with 

910 kind=="fix_attempt", result.passed=True, and a non-empty diff. A 

911 passing baseline/repro attempt never counts, even if it's the most 

912 recent entry.""" 

913 for attempt in reversed(self.attempts): 

914 if attempt.kind == "fix_attempt" and attempt.result.passed and attempt.diff: 

915 return attempt 

916 return None

924 return self._billed_seconds * self._settings.e2b_cost_per_second_usd

937def build_sandbox_tools(handle: SandboxHandle, *, file_read_max_chars: int) -> list[BaseTool]: 

938 """7 `StructuredTool`s as closures over `handle`: read_file, write_file,

949 return short, fixed-shape status strings, not raw file/log content, so 

950 they're left unclamped too."""

1041@asynccontextmanager 

1042async def sandbox_toolset( 

1043 settings: Settings, github_client: Github, repo_full_name: str, ref: str | None = None 

1044) -> AsyncGenerator[tuple[list[BaseTool], SandboxHandle | None]]: 

1045 """Composition-root context manager, mirrors `tools/mcp_clients.py`'s 

1046 `researcher_toolset` pattern but diverges on eagerness: no I/O happens in 

1047 this function's own body -- constructing `SandboxHandle` is a cheap 

1048 no-op (all E2B/PyGithub I/O is inside `ensure_ready()`, called lazily by 

1049 the tool methods on first real use). Yields `(tools, handle)` -- unlike 

1050 `researcher_toolset`, the caller (`DrafterSubgraph.finalize()`) needs the 

1051 handle's recorded attempts directly, not just the tools. Yields 

1052 `(tools=[], handle=None)` when `settings.e2b_api_key` is unset, matching 

1053 the `GITHUB_TOKEN`/`TAVILY_API_KEY` graceful-degradation precedent 

1054 elsewhere in this codebase. `finally: await handle.aclose()` (skipped 

1055 when `handle` is `None`) guarantees the sandbox is never leaked even if 

1056 the caller raises.""" 

1057 if settings.e2b_api_key is None: 

1058 log.warning("sandbox_tool_unavailable", reason="E2B_API_KEY not set") 

1059 yield [], None 

1060 return 

1061 

1062 handle = SandboxHandle( 

1063 settings=settings, github_client=github_client, repo_full_name=repo_full_name, ref=ref 

1064 ) 

1065 try: 

1066 tools = build_sandbox_tools( 

1067 handle, file_read_max_chars=settings.drafter_file_read_max_chars 

1068 ) 

1069 yield tools, handle 

1070 finally: 

1071 await handle.aclose()
